    format {
        [(int frame type id), ...(other)]
    }
    frame type ids {
        0 : key frame,
        1 : update frame,
        2 : expand frame,
        3 : add frame,
        4 : toggle visibility,
        5 : wait
    }
    key frame {
        clears all shapes and repopulates
        [0, {type:(str shape type), props:{(shape props)}, clname:(str css class name, optional), name:(str unique id, optional)}]
    }
